\node (drainQ)  [decision, below=2.2cm of C]    {ops непуст?};
\node (drainLP) [decision, below=of drainQ]     {top — это '('?};
\node (errLP)   [terminal, right=2.6cm of drainLP] {Ошибка: лишняя '('};
\node (drainPop)[process,  below=of drainLP]    {output.push(top); \quad ops.pop()};
\node (ret)     [terminal, below=of drainPop]   {Возврат output};

\draw [arrow] (start) -- (init);
\draw [arrow] (init)  -- (loop);
\draw [arrow] (loop)  -- node[left] {Да} (kind);
\draw [arrow] (loop.east) -- node[above, font=\footnotesize] {Нет} (R.west);
\draw [arrow] (R.south) |- ($(drainQ.east)+(0.2,0)$);

\draw [arrow] (kind.west)  -| node[above, pos=0.2, font=\footnotesize] {Number}     (A);
\draw [arrow] (kind.south) -| node[above, pos=0.7, font=\footnotesize] {Function}   (B);
\draw [arrow] (kind.south) -- node[right, font=\footnotesize] {Operator}            (C);
\draw [arrow] (kind.south) -| node[above, pos=0.7, font=\footnotesize] {LeftParen}  (D);
\draw [arrow] (kind.east)  -| node[above, pos=0.2, font=\footnotesize] {RightParen} (E);

\draw [arrow] (drainQ)  -- node[left] {Да} (drainLP);
\draw [arrow] (drainQ.west) -- ++(-2.6,0) node[midway, above, font=\footnotesize] {Нет} |- (ret.west);
\draw [arrow] (drainLP.east) -- node[above, font=\footnotesize] {Да} (errLP);
\draw [arrow] (drainLP) -- node[left] {Нет} (drainPop);
\draw [arrow] (drainPop) -- (ret);
\draw [arrow] (drainPop.east) -- ++(2.6,0) |- (drainQ.east);

\end{tikzpicture}
