\node (chkEmp)  [decision,   below=of skip1]     {pos $\geq$ size?};

\node (errEmp)  [terminal,   right=2.8cm of chkEmp]

\node (chkTail) [decision,   below=of skip2]     {pos $<$ size?};

\node (errTail) [terminal,   right=2.8cm of chkTail]
                                                 {throw \textquotedbl Лишний символ: \textquotedblleft s[pos]\textquotedblright\textquotedbl};

\node (ret)     [terminal,   below=of chkTail]   {Возврат r};

\draw [arrow] (start)   -- (skip1);
\draw [arrow] (skip1)   -- (chkEmp);
\draw [arrow] (chkEmp)  -- node[above] {Да}  (errEmp);
\draw [arrow] (chkEmp)  -- node[left]  {Нет} (expr);
\draw [arrow] (expr)    -- (skip2);
\draw [arrow] (skip2)   -- (chkTail);
\draw [arrow] (chkTail) -- node[above] {Да}  (errTail);
\draw [arrow] (chkTail) -- node[left]  {Нет} (ret);

\end{tikzpicture}
